import java.util.Scanner;
class Main{
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int lastDigit = num % 10;
        int firstDigit = num / 100;
        int sum = firstDigit + lastDigit;
        System.out.println(sum);
        sc.close();
    }
}
