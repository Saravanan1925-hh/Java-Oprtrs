import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double L = sc.nextDouble();
        double W = sc.nextDouble();
        double H = sc.nextDouble();
        double volume = L * W * H;
        System.out.println(volume);
        sc.close();
    }
}
